\begin{align*} \int_a^b f(x)\, dx \amp = \int_a^0 f(x)\, dx+\int_0^b f(x)\, dx \amp (\knowl{./knowl/th_interval_partition.html}{\text{Interval partition}})\\ \amp = -\int_0^af(x)\, dx+\int_0^bf(x)\, dx \amp (\knowl{./knowl/th_order_integration.html}{\text{Order of integration}}) \\ \amp = -(a-\frac{1}{4}a^4)+(b-\frac{1}{4}b^4) \amp (\knowl{./knowl/eg_int_cubic.html}{\text{1.4.4}}) \\ \amp = (b-a)-\frac{1}{4}(b^4-a^4) \end{align*}
