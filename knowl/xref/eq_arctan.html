\begin{align} \arctan(x)=\theta \amp \iff \tan\theta=x \text{ and } -\pi/2\lt \theta\lt \pi/2\tag{1.16.7}\\ \arctan(\tan \theta)\amp =\theta \text{ for all } -\pi/2\lt \theta\lt \pi/2\tag{1.16.8}\\ \tan(\arctan x)\amp =x \text{ for all } x\tag{1.16.9}\\ \lim_{x\to\infty}\arctan x\amp =\pi/2,\hspace{5pt} \lim_{x\to-\infty}\arctan x=-\pi/2\tag{1.16.10} \end{align}
