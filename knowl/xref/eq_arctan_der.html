\begin{align} \frac{d}{dx} \arcsin x=\frac{1}{\sqrt{1-x^2}}\amp \iff \int \frac{1}{\sqrt{1-x^2}}\, dx=\arcsin x+C \amp \text{ (for all \(x\) in \((-1,1)\)) }\tag{1.16.12}\\ \frac{d}{dx} \arccos x=-\frac{1}{\sqrt{1-x^2}}\amp \iff \int \frac{1}{\sqrt{1-x^2}}\, dx=-\arccos x+C \amp \text{ (for all \(x\) in \((-1,1)\)) }\tag{1.16.13}\\ \frac{d}{dx} \arctan x=\frac{1}{1+x^2}\amp \iff \int \frac{1}{1+x^2}\, dx=\arctan x+C \amp \text{ (for all \(x\)) } \tag{1.16.14} \end{align}
